Mixin Code

SCSS

                
@mixin cap-aligned-line-height($line-height: 1.2) {
    $offset-top: ((0 - (60 / 10)) + ($line-height - 1) * (60 / 2)) / 60;    
    $offset-bottom: ((0 - (60 / 10)) + ($line-height - 1) * (60 / 2)) / 60;    

    line-height: $line-height;

    &::before,
    &::after {
        content: '';
        display: block;
        height: 0;
        width: 0;
    }

    &::before {
        margin-bottom: -#{$offset-top}em;
    }

    &::after {
        margin-top: -#{$offset-bottom}em;
    }
}
