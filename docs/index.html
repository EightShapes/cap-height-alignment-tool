Mixin Code

SCSS

                
@mixin text-crop($line-height: 1.2) {

    $offset-top: max((0 + ($line-height - 1.2) * (60 / 2)), 0) / 60;
    $offset-bottom: max((0 + ($line-height - 1.2) * (60 / 2)), 0) / 60;

    line-height: $line-height;

    &::before,
    &::after {
        content: '';
        display: block;
        height: 0;
        width: 0;
    }

    &::before {
        margin-bottom: -#{$offset-top}em;
    }

    &::after {
        margin-top: -#{$offset-bottom}em;
    }
}
